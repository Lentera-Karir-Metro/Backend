#!/bin/bash
# ========================================
# Supabase Bucket Setup Checklist
# ========================================
# 
# Pastikan 4 bucket ini sudah dibuat di Supabase:
# 
# 1. BUCKET: "thumbnails"
#    - Public access: YES (pilih "Public")
#    - Purpose: Untuk menyimpan gambar thumbnail
#    - File types: JPEG, PNG, WebP, GIF
#    - Max size: 5MB per file
#
# 2. BUCKET: "videos"
#    - Public access: YES (pilih "Public")
#    - Purpose: Untuk menyimpan file video
#    - File types: MP4, WebM
#    - Max size: 500MB per file
#
# 3. BUCKET: "ebooks"
#    - Public access: YES (pilih "Public")
#    - Purpose: Untuk menyimpan file ebook
#    - File types: PDF
#    - Max size: 50MB per file
#
# 4. BUCKET: "certificates"
#    - Public access: YES (pilih "Public")
#    - Purpose: Untuk menyimpan sertifikat
#    - File types: JPEG, PNG, PDF
#    - Max size: 10MB per file
#
# ========================================
# LANGKAH-LANGKAH DI SUPABASE:
# ========================================
#
# 1. Login ke Supabase Dashboard
# 2. Pilih project Anda
# 3. Go to: Storage (sidebar kiri)
# 4. Klik "+ New Bucket" untuk setiap bucket:
#    - Name: "thumbnails"
#    - Public: ON (toggle)
#    - Create
#
#    - Name: "videos"
#    - Public: ON (toggle)
#    - Create
#
#    - Name: "ebooks"
#    - Public: ON (toggle)
#    - Create
#
#    - Name: "certificates"
#    - Public: ON (toggle)
#    - Create
#
# 5. Verify di .env file:
#    SUPABASE_URL=https://[your-project].supabase.co
#    SUPABASE_KEY=[your-anon-key]
#
# ========================================

echo "✅ Bucket setup checklist berhasil!"
